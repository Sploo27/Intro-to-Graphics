#version 300 es

#ifdef GL_ES
precision highp float;
#endif //GL_ES

uniform sampler2D uTex;

in vec2 vTexcoord;

layout (location = 0) out vec4 rtFragColor;

void main()
{
	//rtFragColor = vec4(vTexcoord, 0.0, 0.0); //first idea
	//vec2 = uTex; //second idea
	
	vec2 uv = vTexcoord;
	vec4 col = texture(uTex, uv);
	
	rtFragColor = col;
	
}
